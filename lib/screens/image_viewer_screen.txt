import 'dart:io';
import 'package:flutter/material.dart';
import 'package:share_plus/share_plus.dart';
import 'package:dio/dio.dart';
import 'package:gal/gal.dart';
import 'package:path_provider/path_provider.dart';
import '../services/object_detection_service.dart';

class ImageViewerScreen extends StatefulWidget {
  final List<String> imageUrls;
  final int initialIndex;
  final String title;

  const ImageViewerScreen({
    super.key,
    required this.imageUrls,
    required this.initialIndex,
    required this.title,
  });

  @override
  State<ImageViewerScreen> createState() => _ImageViewerScreenState();
}

class _ImageViewerScreenState extends State<ImageViewerScreen> {
  late final PageController _pageController;
  late int _currentIndex;
  bool _isProcessing = false;

  final ObjectDetectionService _detectionService = ObjectDetectionService();
  List<Map<String, dynamic>> _detectedObjects = [];
  double? _imgWidth;
  double? _imgHeight;

  bool _isDetecting = false;
  bool _showDetection = false;

  @override
  void initState() {
    super.initState();
    _pageController = PageController(initialPage: widget.initialIndex);
    _currentIndex = widget.initialIndex;
    _detectionService.loadModel();
  }

  @override
  void dispose() {
    _pageController.dispose();
    _detectionService.close();
    super.dispose();
  }

  Future<String?> _downloadFile(String url) async {
    try {
      final tempDir = await getTemporaryDirectory();
      final fileName = 'temp_detect_${DateTime.now().millisecondsSinceEpoch}.jpg';
      final savePath = '${tempDir.path}/$fileName';
      await Dio().download(url, savePath);
      return savePath;
    } catch (e) {
      print("Download error: $e");
      return null;
    }
  }

  void _detectPeople() async {
    if (_isDetecting) return;

    setState(() {
      _isDetecting = true;
      _showDetection = false;
      _detectedObjects = [];
      _imgWidth = null;
      _imgHeight = null;
    });

    try {
      final currentUrl = widget.imageUrls[_currentIndex];
      final path = await _downloadFile(currentUrl);

      if (path != null) {
        final result = await _detectionService.detectPerson(path);

        final objects = result['objects'] as List<Map<String, dynamic>>? ?? [];
        final w = result['width'] as double?;
        final h = result['height'] as double?;

        setState(() {
          _detectedObjects = objects;
          _imgWidth = w;
          _imgHeight = h;
          _showDetection = true;
        });

        if (objects.isEmpty) {
          ScaffoldMessenger.of(context).showSnackBar(
            const SnackBar(
              content: Text('Tidak ada objek signifikan terdeteksi.'),
              duration: Duration(seconds: 2),
            ),
          );
        } else {

          final personCount = objects.where((o) => o['label'].toString().toLowerCase().contains('person')).length;
          final msg = personCount > 0
              ? '$personCount Orang terdeteksi!'
              : '${objects.length} Objek terdeteksi (Bukan Orang).';

          ScaffoldMessenger.of(context).showSnackBar(
            SnackBar(
              content: Text(msg),
              backgroundColor: personCount > 0 ? Colors.green : Colors.orange
            ),
          );
        }

        File(path).delete().ignore();
      }
    } catch (e) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('Gagal mendeteksi: $e'), backgroundColor: Colors.red),
      );
    } finally {
      if (mounted) setState(() => _isDetecting = false);
    }
  }

  void _shareImage() async {
    if (_isProcessing) return;
    setState(() => _isProcessing = true);
    try {
      final currentUrl = widget.imageUrls[_currentIndex];
      final path = await _downloadFile(currentUrl);
      if (path != null) {
        await Share.shareXFiles([XFile(path)], text: 'Rekaman CCTV - ${widget.title}');
      } else {
        await Share.share('Lihat rekaman CCTV ini: $currentUrl');
      }
    } catch (e) {
      ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text('Gagal membagikan: $e')));
    } finally {
      if (mounted) setState(() => _isProcessing = false);
    }
  }

  void _saveImage() async {
    if (_isProcessing) return;
    setState(() => _isProcessing = true);
    try {
      if (!await Gal.hasAccess()) await Gal.requestAccess();
      final currentUrl = widget.imageUrls[_currentIndex];
      final path = await _downloadFile(currentUrl);

      if (path != null) {
        await Gal.putImage(path);
        if (mounted) {
          ScaffoldMessenger.of(context).showSnackBar(
            const SnackBar(content: Text('Tersimpan di Galeri!'), backgroundColor: Colors.green),
          );
        }
        final file = File(path);
        if (await file.exists()) await file.delete();
      }
    } catch (e) {
      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text('Gagal menyimpan: $e'), backgroundColor: Colors.red));
      }
    } finally {
      if (mounted) setState(() => _isProcessing = false);
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: Colors.black,
      appBar: AppBar(
        backgroundColor: Colors.black.withOpacity(0.5),
        foregroundColor: Colors.white,
        title: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Text('Pukul ${widget.title}', style: const TextStyle(fontSize: 16)),
            Text(
              '${_currentIndex + 1} dari ${widget.imageUrls.length}',
              style: const TextStyle(fontSize: 12, fontWeight: FontWeight.normal),
            ),
          ],
        ),
        actions: [
          IconButton(
            icon: _isDetecting
                ? const SizedBox(width: 20, height: 20, child: CircularProgressIndicator(color: Colors.white, strokeWidth: 2))
                : Icon(Icons.person_search, color: _showDetection ? Colors.greenAccent : Colors.white),
            onPressed: _detectPeople,
            tooltip: 'Deteksi Orang',
          ),
          IconButton(icon: const Icon(Icons.share), onPressed: _shareImage),
          IconButton(icon: const Icon(Icons.download), onPressed: _saveImage),
        ],
      ),
      extendBodyBehindAppBar: true,
      body: PageView.builder(
        controller: _pageController,
        itemCount: widget.imageUrls.length,
        onPageChanged: (index) {
          setState(() {
            _currentIndex = index;
            _showDetection = false;
            _detectedObjects = [];
            _imgWidth = null;
            _imgHeight = null;
          });
        },
        itemBuilder: (context, index) {
          return InteractiveViewer(
            panEnabled: true,
            minScale: 0.5,
            maxScale: 4.0,
            child: Center(
              child: Stack(
                fit: StackFit.expand,
                children: [

                  Hero(
                    tag: widget.imageUrls[index],
                    child: Image.network(
                      widget.imageUrls[index],
                      fit: BoxFit.contain,
                      loadingBuilder: (ctx, child, progress) {
                        if (progress == null) return child;
                        return Center(child: CircularProgressIndicator(value: progress.expectedTotalBytes != null ? progress.cumulativeBytesLoaded / progress.expectedTotalBytes! : null, color: Colors.white));
                      },
                    ),
                  ),


                  if (_showDetection && _currentIndex == index && _imgWidth != null && _imgHeight != null)
                    CustomPaint(
                      painter: DetectionPainter(
                        objects: _detectedObjects,
                        imgWidth: _imgWidth!,
                        imgHeight: _imgHeight!,
                      ),
                    ),
                ],
              ),
            ),
          );
        },
      ),
    );
  }
}

class DetectionPainter extends CustomPainter {
  final List<Map<String, dynamic>> objects;
  final double imgWidth;
  final double imgHeight;

  DetectionPainter({
    required this.objects,
    required this.imgWidth,
    required this.imgHeight,
  });

  @override
  void paint(Canvas canvas, Size size) {
    if (objects.isEmpty) return;

    double screenAspectRatio = size.width / size.height;
    double imageAspectRatio = imgWidth / imgHeight;

    double drawWidth, drawHeight, dx, dy;

    if (imageAspectRatio > screenAspectRatio) {
      drawWidth = size.width;
      drawHeight = size.width / imageAspectRatio;
      dx = 0;
      dy = (size.height - drawHeight) / 2;
    } else {
      drawHeight = size.height;
      drawWidth = size.height * imageAspectRatio;
      dy = 0;
      dx = (size.width - drawWidth) / 2;
    }

    final imageDrawRect = Rect.fromLTWH(dx, dy, drawWidth, drawHeight);

    for (var obj in objects) {
      final rectData = obj['rect'];
      final confidence = obj['confidence'] as double;
      final label = obj['label'].toString().toLowerCase();





      Color boxColor;
      if (label.contains('person')) {
        boxColor = confidence > 0.7 ? Colors.greenAccent : Colors.yellowAccent;
      } else {
        boxColor = Colors.deepOrangeAccent;
      }

      final paint = Paint()
        ..color = boxColor
        ..style = PaintingStyle.stroke
        ..strokeWidth = 3.0;

      final rect = Rect.fromLTWH(
        dx + (rectData['x'] * drawWidth),
        dy + (rectData['y'] * drawHeight),
        rectData['w'] * drawWidth,
        rectData['h'] * drawHeight,
      );

      final validRect = rect.intersect(imageDrawRect);

      if (!validRect.isEmpty) {
        canvas.drawRect(validRect, paint);


        final textSpan = TextSpan(
          text: "${obj['label']} ${(confidence * 100).toStringAsFixed(0)}%",
          style: const TextStyle(
            color: Colors.black,
            fontSize: 12,
            fontWeight: FontWeight.bold
          ),
        );

        final textPainter = TextPainter(
          text: textSpan,
          textDirection: TextDirection.ltr
        );
        textPainter.layout();

        double labelX = validRect.left;
        double labelY = validRect.top - 20;

        if (labelY < dy) labelY = validRect.top + 5;
        if (labelX < dx) labelX = validRect.left + 5;

        final bgRect = Rect.fromLTWH(
          labelX - 2,
          labelY - 2,
          textPainter.width + 4,
          textPainter.height + 4
        );

        canvas.drawRect(bgRect, Paint()..color = boxColor);
        textPainter.paint(canvas, Offset(labelX, labelY));
      }
    }
  }

  @override
  bool shouldRepaint(covariant CustomPainter oldDelegate) => true;
}
